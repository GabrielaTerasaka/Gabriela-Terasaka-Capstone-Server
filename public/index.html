"/signup"

POST:

REQUEST body { first_name, last_name, password, email } RESPONSE {token}

"/login"

GET:

REQUEST body { password, email } RESPONSE {token}

"/users"

GET:

REQUEST headers { token } RESPONSE all-users [{ id, first_name, last_name, email }]

"/users"

PUT:

REQUEST headers { token } & body { first_name, last_name, email: emailBody } RESPONSE {token}

"/users/:email"

GET:

REQUEST headers { token } RESPONSE { id, first_name, last_name, email }
